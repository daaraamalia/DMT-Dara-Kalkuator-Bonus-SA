      function hitungSemua(){
      const hari = +document.getElementById("hari").value || 0;
      const vol = +document.getElementById("volume").value || 0;
      const vPremi = +document.getElementById("vP").value || 0;
      const sPremi = +document.getElementById("sP").value || 0;
      const potRate = +document.getElementById("nl").value;
      const targetMinimal = +document.getElementById("vintage").value;
      if(!hari) return alert("Isi hari dulu");
      const gDasar = (+document.getElementById("is").value) * hari;
      let isZonk = vol < targetMinimal;
      let bNOS = (
      (+document.getElementById("FNC").value * 100000 || 0) +
      (+document.getElementById("FRO").value * 50000 || 0) +
      (+document.getElementById("NNC").value * 80000 || 0) +
      (+document.getElementById("NRO").value * 40000 || 0)
      );
      let bVAS = vPremi * 0.04;
      let bSAI = sPremi * 0.07;
      const gross = bNOS + bVAS + bSAI;
      const pot = gross * potRate;
      const total = gDasar + gross - pot;
      const res = document.getElementById("hasil");
      res.style.display="block";
      res.innerHTML=`
      <h3>HASIL BONUS</h3>
      <div>Gaji: Rp ${gDasar.toLocaleString()}</div>
      <div>Bonus: Rp ${gross.toLocaleString()}</div>
      <div>Potongan: Rp ${pot.toLocaleString()}</div>
      <hr>
      <h2>Total: Rp ${total.toLocaleString()}</h2>
      `;
      }
